\section{Motivation}

\begin{frame}{Why another theme?}
  \begin{itemize}
    \item Inspired by THU and UCASS beamer themes
    \item Same clean look: \texttt{smoothbars} + \texttt{circles} + custom footline
    \item No institutional branding, no hardcoded logo
    \item One accent color drives everything
  \end{itemize}
\end{frame}

\section{Customization}

\begin{frame}[fragile]{Changing color}
  Pass an option to \verb|\usetheme|:
  \begin{verbatim}
\usetheme[color=red]{neat}      % preset
\usetheme[accent=1F497D]{neat}  % custom hex
\usetheme[sectoc=false]{neat}   % no auto-TOC
  \end{verbatim}
  Presets: \texttt{blue} (default), \texttt{red}, \texttt{purple},
  \texttt{green}, \texttt{teal}, \texttt{orange}, \texttt{gray}.
\end{frame}
